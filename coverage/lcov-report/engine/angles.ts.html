export function parseDMSString(input: string): number | null {
  // Handle formats: 45°30'22.5", 45 30 22.5, 45.5, N45°30'E
  const cleaned = input.replace(/[°'"]/g, ' ').trim().replace(/\s+/g, ' ');
  const parts = cleaned.split(' ');
  
  if (parts.length === 1) {
